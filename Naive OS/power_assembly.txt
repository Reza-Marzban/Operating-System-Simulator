		POP,	51		SAVE INPUTS TO MEMORY

		POP,	49		SAVES THE FIRST INPUT IN 49
L1:		PUSH,	50		LABEL1
		MUL,	49		THE FIRST INPUT STORED IN 49 IS MULTIPLIED BY TOTAL IN 50
		POP,	50		SAVE THE NEW TOTAL IN 50
		PUSH,	51		LOADS THE SECOND INPUT
		PUSH,	ONE
		SUB,			DECREMENT THE SECOND INPUT BY ONE
		POP,	51		SAVES THE NEW VALUE
		PUSH,	ZERO
		CPL,	51		COMPARE THE SECOND INPUT WITH ZERO
		BRT,	L1		IF LESS THAN ZERO, IT WILL GOTO LABEL1
		PUSH,	50		PUSH THE FINAL VALUE
