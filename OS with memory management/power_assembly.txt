START	RD,				FIRST_INPUT^SECOND_INPUT
	RD,				GET TWO INPUTS
	POP,	A		SAVE INPUTS TO MEMORY
	POP,	B
	PUSH,	B		
	POP,	C		SAVES THE FIRST INPUT IN 49
L1:	PUSH,	B		LABEL1
	MUL,	C		THE FIRST INPUT STORED IN 49 IS MULTIPLIED BY TOTAL IN 50
	POP,	B		SAVE THE NEW TOTAL IN 50
	PUSH,	A		LOADS THE SECOND INPUT
	PUSH,	ONE
	SUB,			DECREMENT THE SECOND INPUT BY ONE
	POP,	A		SAVES THE NEW VALUE
	PUSH,	ZERO
	CPL,	A		COMPARE THE SECOND INPUT WITH ZERO
	BRT,	L1		IF LESS THAN ZERO, IT WILL GOTO LABEL1
	PUSH,	B		PUSH THE FINAL VALUE
	WR				PRINT THE FINAL VALUE
	HLT				PROGRAM TERMINATES
